git init    -   создает репозиторий
git remote add название ссылка на ссн ключ
git remote -v  -  посмотреть все сохраненные ссн ключи
git add .   -   просто пишем эту команду
git commit -m 'massage'     - cоздаем новую версию 
git push origin master  -   закидываем версию на гидхаб
git pill origin master   -  стягиваем новые данные с гидхаба
git clone ssh-cсылка  -  копируем репозиторий с гидхаба(можно чужой)
git log  -  посмотреть все коммиты
